  <!-- JAVASCRIPT INTEGRADO (Lógica Web3, Fetch, Try/Catch e Manipulação do DOM) -->
  <script>
    /**
     * CONFIGURAÇÃO E CONSTANTES
     * Substitua pela sua chave JWT real da Pinata para testes em produção ou informe pelo formulário.
     */
    const PINATA_JWT_PADRAO = ""; 
    const PINATA_API_ENDPOINT = "https://api.pinata.cloud/pinning/pinJSONToIPFS";

    // Elementos do DOM
    const formEvento = document.getElementById('formEvento');
    const userNameInput = document.getElementById('userName');
    const actionNameInput = document.getElementById('actionName');
    const criticalitySelect = document.getElementById('criticality');
    const pinataJwtInput = document.getElementById('pinataJwt');
    const btnSubmit = document.getElementById('btnSubmit');
    const statusArea = document.getElementById('statusArea');
    const logsList = document.getElementById('logsList');
    const emptyState = document.getElementById('emptyState');

    /**
     * REQUISITO 3: Função para envio e gravação no IPFS via Pinata API
     * Utiliza async/await e a Fetch API nativa.
     */
    async function registrarLogNoIPFS(dadosEvento, jwtTokenUser) {
      const tokenAtivo = jwtTokenUser?.trim() || PINATA_JWT_PADRAO.trim();

      // Se não houver token configurado, executa simulação para teste local seguro
      if (!tokenAtivo) {
        // Simula delay de rede descentralizada (1.5s)
        await new Promise(resolve => setTimeout(resolve, 1500));
        
        // Retorna Hash CID fictício com padrão IPFS
        const randomCid = "bafkrei" + Array.from({length: 32}, () => Math.floor(Math.random()*16).toString(16)).join('');
        return { IpfsHash: randomCid, isSimulated: true };
      }

      // REQUISITO 3: Montagem da requisição com pinataContent e Authorization Bearer
      const payloadPinata = {
        pinataContent: {
          ...dadosEvento,
          registradoEmUTC: new Date().toISOString()
        },
        pinataMetadata: {
          name: `LOG_AUDITORIA_${dadosEvento.criticality}_${Date.now()}`
        }
      };

      // Chamada HTTP POST nativa com Fetch API
      const response = await fetch(PINATA_API_ENDPOINT, {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'Authorization': `Bearer ${tokenAtivo}`
        },
        body: JSON.stringify(payloadPinata)
      });

      // Verificação de erro HTTP (Ex: 401 Unauthorized)
      if (!response.ok) {
        if (response.status === 401) {
          throw new Error('Falha de Autenticação (401): O Token JWT do Pinata é inválido ou expirou.');
        }
        throw new Error(`Erro na API Pinata (${response.status}): ${response.statusText}`);
      }

      const data = await response.json();
      return { IpfsHash: data.IpfsHash, isSimulated: false };
    }

    /**
     * REQUISITO 2 e 4: Handler de submissão com event.preventDefault() e try/catch
     */
    formEvento.addEventListener('submit', async (e) => {
      // 1. Impede a recarga da página (no refresh)
      e.preventDefault();

      // Extração dos valores dos campos do DOM
      const dadosEvento = {
        user: userNameInput.value.trim(),
        action: actionNameInput.value.trim(),
        criticality: criticalitySelect.value,
        timestampISO: new Date().toISOString()
      };

      const userJwt = pinataJwtInput.value.trim();

      // REQUISITO 2: Atualiza o DOM para o estado "Registrando na rede IPFS..."
      statusArea.className = "status-msg loading";
      statusArea.innerText = "Registrando na rede IPFS... Aguarde a prova criptográfica (CID).";
      btnSubmit.disabled = true;

      // REQUISITO 4: Tratamento de Erros via Bloco Try/Catch
      try {
        // Chamada assíncrona
        const resultado = await registrarLogNoIPFS(dadosEvento, userJwt);

        // REQUISITO 5: Atualização dinâmica do DOM em caso de sucesso
        renderizarCardAudit(dadosEvento, resultado.IpfsHash, resultado.isSimulated);

        // Feedback visual verde no DOM
        statusArea.className = "status-msg success";
        statusArea.innerText = `Evento gravado com sucesso no IPFS! CID: ${resultado.IpfsHash}`;

        // Limpa campos para o próximo registro
        actionNameInput.value = '';

      } catch (erro) {
        // REQUISITO 4: Em caso de falha, exibe mensagem vermelha de erro no DOM
        console.error("Erro ao registrar evento:", erro);
        statusArea.className = "status-msg error";
        statusArea.innerText = ` [ERRO]: ${erro.message || 'Falha de conexão com o nó IPFS.'}`;
      } finally {
        btnSubmit.disabled = false;
      }
    });

    /**
     * REQUISITO 5: Renderização Dinâmica do Card de Auditoria no DOM
     */
    function renderizarCardAudit(dados, ipfsHash, isSimulated) {
      if (emptyState) {
        emptyState.style.display = 'none';
      }

      const card = document.createElement('div');
      card.className = `audit-card ${dados.criticality}`;

      card.innerHTML = `
        <div class="card-header-row">
          <span class="card-user">${escapeHtml(dados.user)}</span>
          <span class="card-time">${new Date().toLocaleString()}</span>
        </div>
        <div class="card-action">${escapeHtml(dados.action)}</div>
        <div style="font-size:11px; color:var(--text-muted);">
          Nível de Criticidade: <strong>${dados.criticality}</strong> ${isSimulated ? '(Simulado)' : '(Pinata Gateway)'}
        </div>
        <div class="cid-badge">
          PROVA CID (IPFS): ${ipfsHash}
        </div>
      `;

      // Adiciona o novo card no topo da lista sem recarregar a tela
      logsList.prepend(card);
    }

    // Auxiliar para segurança contra XSS no DOM
    function escapeHtml(text) {
      return text.replace(/[&<>"']/g, function(m) {
        return { '&': '&amp;', '<': '&lt;', '>': '&gt;', '"': '&quot;', "'": '&#039;' }[m];
      });
    }
  </script>
